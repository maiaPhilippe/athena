query($number_of_repos:Int! $next:String, $org:String!)  {
  organization(login: $org) {
    repositories(first: $number_of_repos, after:$next) {
      edges {
        node {
          defaultBranchRef{
            target{
              repository{
                licenseInfo {
                 licenseId: id
                  licenseType: key
                }
              }
            }
          }
          id
          name
          description
          url
          nameWithOwner
          isPrivate
          primaryLanguage {
            priLanguage: name
          }
          forks{
            totalCount
          }
          stargazers{
            totalCount
          }
          watchers{
             totalCount
          }
          createdAt
        }
        cursor
      }
    }
  }
}
