query ($number_of_repos: Int!, $next: String, $org: String!) {
  organization(login: $org) {
    teams(first: $number_of_repos, after: $next) {
      totalCount
      pageInfo {
        endCursor
        hasNextPage
      }
      edges {
        node {
          members(first:100) {
            membersCount: totalCount
            edges {
              node {
                memberId: id
                memberName:name
              }
            }
          }
          repositories(first:100) {
            repoCount: totalCount
            edges {
              node {
                repoId:id
               repoName: name
              }
            }
          }
          createdAt
          id
          name
          privacy
          slug
        }
      }
    }
  }
rateLimit{
  limit
  remaining
}
}
